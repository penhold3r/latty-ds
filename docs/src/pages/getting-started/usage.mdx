# Usage

## Vanilla HTML

The fastest way to get started — no build tool required:

```html
<!DOCTYPE html>
<html lang="en">
<head>
  <link rel="stylesheet" href="node_modules/@latty/tokens/dist/tokens.css">
</head>
<body>
  <lt-button variant="primary" icon="save">Save changes</lt-button>

  <script type="module">
    import '@latty/web';
  </script>
</body>
</html>
```

## With a bundler (Vite, Webpack, etc.)

```js
// entry.js
import '@latty/tokens/dist/tokens.css';
import '@latty/web';
```

```html
<lt-button variant="primary">Click me</lt-button>

<lt-textfield
  label="Email address"
  type="email"
  required
  helper-text="We'll never share your email"
></lt-textfield>
```

## React

Use the `@latty/react` wrapper package for typed components and idiomatic React events:

```tsx
import { Button, Textfield } from '@latty/react';

export function Example() {
  return (
    <Button variant="primary" icon="save" onClick={() => console.log('saved')}>
      Save changes
    </Button>
  );
}
```

## Vue

Vue 3 supports Web Components natively. Add one-time configuration in your app entry:

```ts
// main.ts
import { createApp } from 'vue'
import '@latty/tokens/dist/tokens.css'
import '@latty/web'
import App from './App.vue'

const app = createApp(App)
app.config.compilerOptions.isCustomElement = (tag) => tag.startsWith('lt-')
app.mount('#app')
```

Then use `lt-*` elements directly in any component template. Use Vue's `@event` syntax for listeners and `:prop` for reactive bindings:

```vue
<script setup lang="ts">
function save() { /* ... */ }
function onEmail(e: Event) {
  console.log((e as CustomEvent).detail.value)
}
</script>

<template>
  <lt-button variant="primary" @click="save">Save changes</lt-button>
  <lt-textfield label="Email" type="email" @lt-change="onEmail"></lt-textfield>
</template>
```

No wrapper package needed — `lt-*` elements work natively in Vue templates.

## Listening to events

Components dispatch standard DOM events:

```js
const dialog = document.querySelector('lt-dialog');

dialog.addEventListener('lt-open', () => console.log('opened'));
dialog.addEventListener('lt-close', () => console.log('closed'));
```

```js
const textfield = document.querySelector('lt-textfield');

textfield.addEventListener('lt-change', (e) => {
  console.log(e.detail.value);
});
```

## Setting properties programmatically

Attributes handle strings and booleans. For richer data (e.g. table columns), use the JS property directly:

```js
const table = document.querySelector('lt-table');
table.columns = [
  { key: 'name', label: 'Name' },
  { key: 'email', label: 'Email' },
];
table.data = [
  { name: 'Alice', email: 'alice@example.com' },
];
```
